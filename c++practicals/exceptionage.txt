#include <iostream>
using namespace std;

int main()
{
    int age;
    cout<<"Enter your age:"<<endl;
    cin>>age;
    try {
        if (age<0 || age>100) {
            throw (age);
        }
        cout<<"your age is:"<<age<<endl;
    }
    catch(int var) {
        cout<<"Exception:Invalid age"<<endl;
    }
    return 0;
    
}
