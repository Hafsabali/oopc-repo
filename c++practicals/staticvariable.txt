#include <iostream>
using namespace std;
static int count=0;
class StaticDemo 
{public:
    static int destroyedcount;
    StaticDemo() {
        count++;
    }
    static void dispaly() {
        cout<<"Number of objects created:"<<count<<endl;
    }
    ~StaticDemo() {
        count--;
        destroyedcount++;
        
    }
    static void dispalyDistroyed() {
        cout<<"Number of object destroyed:"<<destroyedcount<<endl;
    }
};
int StaticDemo::destroyedcount=0;
int main() 
{
    StaticDemo s1;
    StaticDemo s2;
    StaticDemo s3;
    s3.dispaly();
    StaticDemo::dispalyDistroyed();
    
}
