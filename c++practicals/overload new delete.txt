#include <iostream>
using namespace std;
class Myclass
{public:
    
    int data;
    Myclass(int val) {
        data=val;
        cout<<"Constructor: object created with data="<<data<<endl;
    }
    ~Myclass() {
        cout<<"Object with data="<<data<<" destroyed"<<endl;
    }
    void* operator new(size_t size) {
        cout<<"Overloaded new: Allocating "<<size<<" bytes"<<endl;
        void *ptr =::operator new(size);
        return ptr;
    }
    void operator delete(void* ptr) {
        cout<<"Overloaded delete:Deallocating memory"<<endl;
        ::operator delete(ptr);
    }
};


int main()
{
    Myclass* obj1=new Myclass(10);
    Myclass* obj2=new Myclass(20);
    delete obj1;
    delete obj2;
    return 0;
}
