#include <iostream>
using namespace std;

int main()
{
    int numerator,denominator,result;
    cout<<"Enter numerator and denominator"<<endl;
    cin>>numerator>>denominator;
    
    try {
        if (denominator==0) {
          throw denominator;
        }
        result=numerator/denominator;
        cout<<"Division is:"<<result;
    }
    catch(int exc) {
         cout<<"Exception: Divide by zero is not allowed!"<<endl;
    }
    
    return 0;
}
